(
  (cond
    (
      (= RELIGION "Moon Daughter")
      (def lines '("Greetings, friend. Hail to the night. Hail to the shadows."))
    )

    (
      (and (<= FAITH 1) (> PLAYER_DEPTH 0) (= DIALOGUE_STATE 0))

      (
        (def lines '("You've been into the darkness. Embracing the shadows might be your best way forward."
                    "Wasn't the moon lovely last night?"
                    "Sometimes I feel more safe, more at peace, in the night."
        ))

        (option "What do you mean?" ( (set DIALOGUE_STATE 1)))
        (option "Farewell." (end "Fare thee well."))
      )
    )

    (
      (and (<= FAITH 1) (= DIALOGUE_STATE 1))

      (
        (def lines '("Hunktokar isn't the only deity who seeks to aid mortals."
                    "Perhaps the best approach isn't always to charge headlong into battle."
                    "There is a path you could take, guided by moonlight."
        ))

        (option "Tell me more." ((set DIALOGUE_STATE 2) (set MD_CURIOUS true) (check-md-cleric) (end "Seek the Standing Stones at night.")))
        (option "Farewell." (end "Fare thee well."))
      )
    )

    (
      (and (<= FAITH 1) (= DIALOGUE_STATE 2))

      (
        (def lines '("Seek the priestess for guidance."
                    "If you truly wish to embrace the moonlit path, seek out our teacher."
                    "The dark needn't be your enemy."
                    "Seek the Standing Stones at night."
        ))
      )
    )

    (
      else (
        (def lines '("#TOWN_NAME is a lovely hamlet, no?"))
      )
    )
  )
  
  (say (pick lines))
)
